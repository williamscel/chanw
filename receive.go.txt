package main
 
import (
    "fmt"
    "net"
    "os"
)
 
//check for error
func CheckError(err error) {
    if err  != nil {
        fmt.Println("Error: " , err)
        os.Exit(0)
    }
}
 
func main() {
    //send to any address at port 10001   
    ReceiveAddr,err := net.ResolveUDPAddr("udp",":10001")
    CheckError(err)
 
    //listen on the port
    ReceiveConn, err := net.ListenUDP("udp", ReceiveAddr)
    CheckError(err)
    defer ReceiveConn.Close()
 
    buf := make([]byte, 1024)
 
    for {
        n,addr,err := ReceiveConn.ReadFromUDP(buf)
        fmt.Println("Received ",string(buf[0:n]), " from ",addr)
 
        if err != nil {
            fmt.Println("Error: ",err)
        } 
    }
}
